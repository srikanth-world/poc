using System.Web.Http;
using System.Web.Mvc;

namespace BAML.GMOT.CHILD.Reporting_Web.Areas.GTS
{
    public class GTSAreaRegistration : AreaRegistration
    {
        public override string AreaName
        {
            get { return "GTS"; }
        }

        public override void RegisterArea(AreaRegistrationContext context)
        {
            // MVC Routing
            context.MapRoute(
                "GTS_default",
                "GTS/{controller}/{action}/{reportId}",
                new { action = "Index", reportId = UrlParameter.Optional },
                new[] { "BAML.GMOT.CHILD.Reporting_Web.Areas.GTS.Controllers" }
            );

            // API Routing
            RegisterApiRoutes(context);
        }

        private void RegisterApiRoutes(AreaRegistrationContext context)
        {
            // Get the HttpConfiguration for the GTS area
            var config = GlobalConfiguration.Configuration;

            // Define API routes
            config.Routes.MapHttpRoute(
                name: "GTS_Api",
                routeTemplate: "GTS/api/{controller}/{action}/{id}",
                defaults: new { id = RouteParameter.Optional }
            );
        }
    }
}
