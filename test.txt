import { Component } from '@angular/core';
import { HttpClient } from '@angular/common/http';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.css']
})
export class AppComponent {
  isLoading: boolean = false; // Loading flag
  results: any[] = []; // Data from API

  constructor(private http: HttpClient) {}

  onSearch() {
    this.isLoading = true; // Show spinner

    // Example API call
    this.http.get<any[]>('https://api.example.com/data').subscribe(
      (data) => {
        this.results = data; // Assign data to results
        this.isLoading = false; // Hide spinner
      },
      (error) => {
        console.error('Error fetching data:', error);
        this.isLoading = false; // Hide spinner even if there's an error
      }
    );
  }
}

<div class="container mt-4">
  <div class="row justify-content-center">
    <div class="col-md-12">
      <form (ngSubmit)="onSearch()" class="mb-4">
        <!-- Your form fields here -->
        <button type="submit" class="btn btn-primary">Search</button>
      </form>

      <!-- Loading Spinner -->
      <div *ngIf="isLoading" class="text-center">
        <div class="spinner-border text-primary" role="status">
          <span class="visually-hidden">Loading...</span>
        </div>
        <p class="mt-2">Loading data...</p>
      </div>

      <!-- Results Table -->
      <div *ngIf="results && !isLoading" class="table-responsive">
        <table class="table table-striped table-hover">
          <!-- Table content here -->
        </table>
      </div>
    </div>
  </div>
</div>
